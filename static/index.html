            Integrate in minutes
          

          

            Use MCP with your AI assistant or call the REST API directly. Here is everything you need to get started.
          

        


        

          
          

            

              
              MCP Configuration
            

            

              
              // claude_desktop_config.json
{
  "mcpServers": {
    "quantum-relayer": {
      "command": "uvx",
      "args": ["quantum-mcp-relayer"],
      "env": {
        "QUANTUM_API_KEY": "your-api-key"
      }
    }
  }
}
            

          


          
          

            

              
              cURL Example
            

            

              
              # Optimize a portfolio with quantum QAOA
curl -X POST http://localhost:8000/api/v1/portfolio/optimize \
  -H "Content-Type: application/json" \
  -d '{
    "assets": [
      {"symbol": "AAPL", "expected_return": 0.12, "volatility": 0.22},
      {"symbol": "GOOGL", "expected_return": 0.15, "volatility": 0.28},
      {"symbol": "BTC", "expected_return": 0.45, "volatility": 0.65}
    ],
    "risk_tolerance": 0.6
  }'
            

          


          
          

            

              
              Python SDK
            

            

              
              import httpx

response = httpx.post(
    "http://localhost:8000/api/v1/route/optimize",
    json={
        "locations": [
            {"name": "SF", "lat": 37.77, "lon": -122.42},
            {"name": "LA", "lat": 34.05, "lon": -118.24},
            {"name": "SEA", "lat": 47.61, "lon": -122.33},
        ],
        "return_to_start": True
    }
)
result = response.json()
print(f"Optimal route: {result['route']}")
print(f"Distance: {result['total_distance_km']:.1f} km")
